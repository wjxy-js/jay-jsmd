function sum() {

    let total = 0;

    for (let i = 0; i < arguments.length; i++) {

        let item = Number(arguments[i]);

        if (isNaN(item)) continue;

        total += item;

    }

    return total;

}

let result = sum(10, 20, 30, '40', 'AA');
